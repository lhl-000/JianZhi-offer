// 从后到前， 顺序排序， 每次循环从后开始将最小值放在前面
public void bubbleSort(int[] arr) 
{
    for (int i=0; i<arr.length; i++) {
        for (int j=arr.length-1; j>i; j-->) {
            if (arr[j-1] > arr[j]) {
                swap(arr, j-1, j);
            }
        }
    }
}

public void bubbleSort(int[] arr) 
{
    boolean isLoop = true;
    for (int i=0; i<arr.length && isLoop; i++) {
        for (int j=arr.length-1; j>i; j-->) {
            isloop = false;
            if (arr[j-1] > arr[j]) {
                swap(arr, j-1, j);
                isloop = true;
            }
        }
    }
}

// 从前到后， 顺序排序， 每次循环从头开始将最大值放倒最后
public void bubbleSort(int[] arr) 
{
    for (int i=0; i<arr.length; i++) {
        for (int j=0; j<arr.length-i-1; j++>) {
            if (arr[j] > arr[j+1]) {
                swap(arr, j, j+1);
            }
        }
    }
}

public void swap(int[] arr, int i, int j) 
{
    int tmp  = i;
    arr[i] = arr[j];
    arr[j] = arr[i];
}
