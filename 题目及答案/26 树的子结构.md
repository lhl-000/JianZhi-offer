 # 25 合并两个排序的链表

### 题目

输入两棵二叉树A和B，判断B是不是A的子结构。(约定空树不是任意一个树的子结构)

B是A的子结构， 即 A中有出现和B相同的结构和节点值。


### 解答


##### 1. 深度优先搜索+递归
```
/JS/
var isSubStructure = function(A, B) {
    var recall = function(A, B) {
        if (B == null) {
            return true;
        }
        if (A == null) {
            return false;
        }
        if (A.val != B.val) {
            return false;
        }
        return recall(A.left, B.left) && recall(A.right, B.right)
    }
    return (A != null && B != null && ( recall(A, B) || isSubStructure(A.left, B) || isSubStructure(A.right, B)));

/JAVA/
class Solution {
    public boolean isSubStructure(TreeNode A, TreeNode B) {
        return (A != null && B != null) && (recall(A,B) || isSubStructure(A.left, B) || isSubStructure(A.right, B));
    }
    public boolean recall (TreeNode A, TreeNode B) {
        if (B == null) {
            return true;
        }
        if (A == null || A.val != B.val) {
            return false;
        }
        return recall(A.left, B.left) && recall(A.right, B.right);
    }
}
```
