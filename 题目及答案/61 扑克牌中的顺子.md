## 61 扑克牌中的顺子

### 题目
从若干副扑克牌中随机抽 5 张牌，判断是不是一个顺子，即这5张牌是不是连续的。2～10为数字本身，A为1，J为11，Q为12，K为13，而大、小王为 0 ，可以看成任意数字。A 不能视为 14。

### 解答


##### 1. 设计
```
/JS/
var isStraight = function(nums) {
    if (nums == null || nums.length != 5) {
        return false;
    }
    nums.sort((a,b)=>{
        return a-b;
    });
    let pre = -1;
    let k = 0;
    for (let i=0; i<5; i++) {
       if (nums[i] == 0) {
           k++;
           continue;
       } else {
           if (pre == -1) {
               pre = nums[i];
               continue;
           }
            if (pre + 1 != nums[i]) {
                if (nums[i] > pre && nums[i] - pre - 1 <= k) {
                    k -= nums[i] -pre -1;
                    pre = nums[i];
                    continue;}
                    return false;
            } else {
                pre++;
            }
       }
    }
    return true;
};
/改进/
var isStraight = function(nums) {
    if (nums == null || nums.length != 5) {
        return false;
    }
    nums.sort((a,b)=>{
        return a-b;
    });
    let k = 0;
    for (let i=0; i<4; i++) {
       if (nums[i] == 0) {
           k++;
           continue;
       }
       if (nums[i] == nums[i+1]) {
           return false;
       }
    }
    return nums[4] - nums[k] < 5;
};
```

##### 2. 集合 Set + 遍历
```
/JS/
class Solution {
    public boolean isStraight(int[] nums) {
        Set<Integer> set = new HashSet<>();
        int max = 0;
        int min = 14;
        for (int i=0; i<5; i++) {
            if (nums[i] == 0) {
                continue;
            }
            max = Math.max(max, nums[i]);
            min = Math.min(min, nums[i]);
            if (set.contains(nums[i])) {
                return false;
            }
            set.add(nums[i]);
        }
        return max - min < 5; 
    }
}
```
