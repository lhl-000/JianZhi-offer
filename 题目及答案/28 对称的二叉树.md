 # 25 二叉树的镜像

### 题目

请实现一个函数，用来判断一棵二叉树是不是对称的。如果一棵二叉树和它的镜像一样，那么它是对称的。

### 解答


##### 1. 二叉树的镜像
```
/JS/
var isSymmetric = function(root) {
    if (root == null) {
        return true;
    }
    var recall = function(A, B) {
        if (A == null && B == null) {
            return true;
        }
        if (A == null || B == null || A.val != B.val) {
            return false;
        }
        return recall(A.left, B.right) && recall(A.right, B.left);
    }
    return recall(root.left, root.right);
};
/JAVA/
class Solution {
    public boolean isSymmetric(TreeNode root) {
        if (root == null) {
            return true;
        }
        return recall(root.left, root.right);
    }
    public boolean recall(TreeNode A, TreeNode B) {
        if (A == null && B == null) {
            return true;
        }
        if (A == null || B == null || A.val != B.val) {
            return false;
        } 
        return recall(A.left, B.right) && recall(A.right, B.left);
    }
}
```
